# 응급실 이용자 현황 데이터 전처리
del_col = list(em_use_2019.columns)
use_col = ['구분','성별','계']
for column_name in del_col:
    if column_name in use_col:
        pass
    else:
        em_use_2019 = em_use_2019.drop([column_name],axis = 1) #사용할 열 빼고 전부 삭제

for i in em_use_2019.index:
    if em_use_2019['성별'][i] == '남자' or em_use_2019['성별'][i] == '여자':
           em_use_2019 = em_use_2019.drop([i]) #성별이 남자 or 여자인 행 삭제
            
del em_use_2019['성별'] #성별 열 삭제

em_use_2019.set_index('구분', inplace=True) #index -> 구별로 변경
em_use_2019 = em_use_2019.drop(['타지역', '미상/미입력','합계']) # 안쓰는 행 삭제

for i in range(len(em_use_2019)):
         em_use_2019['계'][i]  = em_use_2019['계'][i].replace(",","") #콤마 삭제

em_use_2019.columns = ['응급실이용인원'] #열 이름 변경
em_use_2019 = em_use_2019.apply(pd.to_numeric, errors='coerce').fillna(0)# Dataframe의 dtype 을 바꿔줌


#응급실 위치정보 데이터 전처리
seoul_em['주소'][0].split(' ')[1]
